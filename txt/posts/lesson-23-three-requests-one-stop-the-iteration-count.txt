  SPECIAL -- LESSON 23: THREE REQUESTS, ONE STOP — THE ITERATION COUNT
  Lesson 23: Three Requests, One Stop — The Iteration Count
  ============================================================================================



  Lesson 22 found 2 iterations for 1 request. Push it to 3 requests and see whether the
  pattern is really N + 1, or whether 1 request was a special case.

  --------
  FIRST, THE TRACE FOR THREE REQUESTS
  experiment/test_three_req_one_stop.cpp. Compile it yourself, no Makefile, no CMake:
```text
g++ -std=c++17 -o demo lesson-23-three-requests-one-stop.cpp
./demo
```
```text
#include <iostream>
#include <thread>
#include <mutex>
#include <condition_variable>
#include <queue>
#include <chrono>
#include <string>
#include <fstream>

class Logger {
public:
    Logger() {
        file_ = fopen("test_three_req.log", "w");
        background_writer_ = std::thread(&Logger::processQueue, this);
    }

    ~Logger() {
        std::cout << "[~Logger] Shutdown started\n";
        {
            std::unique_lock<std::mutex> lock(mtx_);
            stop_ = true;
        }
        cv_.notify_one();
        if (background_writer_.joinable()) background_writer_.join();
        flushPending();
        if (file_) fclose(file_);
        std::cout << "[~Logger] Shutdown done\n";
    }

    void log(const std::string& msg) {
        std::string formatted = msg + "\n";
        {
            std::lock_guard<std::mutex> lock(mtx_);
            log_queue_.push(std::move(formatted));
        }
        cv_.notify_one();
    }

private:
    void processQueue() {
        int iteration = 0;
        while (true) {
            iteration++;
            std::unique_lock<std::mutex> lock(mtx_);
            cv_.wait(lock, [this]() { return stop_ || !log_queue_.empty(); });

            std::cout << "[processQueue] Iteration " << iteration
                      << " -- stop_=" << stop_
                      << " queue_size=" << log_queue_.size()
                      << " pending_size=" << pending_.size() << "\n";

            if (stop_ && log_queue_.empty()) {
                std::cout << "[processQueue] stop && empty -> flushing pending_ \""
                          << (pending_.empty() ? "(empty)" : pending_.substr(0, pending_.size()-1))
                          << "\" and breaking\n";
                flushPending();
                break;
            }

            std::string msg = std::move(log_queue_.front());
            log_queue_.pop();
            lock.unlock();

            pending_ += msg;
            std::cout << "[processQueue] Popped \"" << msg.substr(0, msg.size()-1)
                      << "\" pending_ now " << pending_.size() << " bytes\n";
            if (pending_.size() >= 4096) flushPending();
        }
        std::cout << "[processQueue] Exited loop after " << iteration << " iterations\n";
    }

    void flushPending() {
        if (file_ && !pending_.empty()) {
            std::cout << "[flushPending] Writing to disk: \"" << pending_.substr(0, pending_.size()-1) << "\"\n";
            fprintf(file_, "%s", pending_.c_str());
            fflush(file_);
            pending_.clear();
        } else {
            std::cout << "[flushPending] Nothing to flush\n";
        }
    }

    std::mutex mtx_;
    std::condition_variable cv_;
    std::queue<std::string> log_queue_;
    std::thread background_writer_;
    bool stop_ = false;
    FILE* file_ = nullptr;
    std::string pending_;
};

int main() {
    std::cout << "[main] === TEST: Three requests, then stop ===\n";
    {
        Logger logger;

        std::cout << "[main] Request 1 arrives\n";
        logger.log("Request 1: User logged in");
        std::this_thread::sleep_for(std::chrono::milliseconds(50));

        std::cout << "[main] Request 2 arrives\n";
        logger.log("Request 2: User clicked button");
        std::this_thread::sleep_for(std::chrono::milliseconds(50));

        std::cout << "[main] Request 3 arrives\n";
        logger.log("Request 3: User saved document");
        std::this_thread::sleep_for(std::chrono::milliseconds(50));

        std::cout << "[main] Shutdown signal received\n";
    }

    std::cout << "\n[main] === VERIFYING LOG FILE ===\n";
    std::ifstream log_file("test_three_req.log");
    std::string line;
    int count = 0;
    while (std::getline(log_file, line)) {
        std::cout << "  Line " << ++count << ": \"" << line << "\"\n";
    }
    log_file.close();

    if (count == 3) {
        std::cout << "[main] PASS: Exactly 3 logs written\n";
    } else {
        std::cout << "[main] FAIL: Expected 3 logs, got " << count << "\n";
    }

    std::remove("test_three_req.log");
    return count == 3 ? 0 : 1;
}
```

  --------
  SO THE THREE MESSAGES LAND IN ONE WRITE, NOT THREE
  The 3 log messages accumulate in pending_ and hit disk as a single write:
```text
"Request 1: User logged in
Request 2: User clicked button
Request 3: User saved document"
```
  Three log lines, one fprintf call -- the batch-write optimization pending_ exists for.

  --------
  SO THE LOOP TURNS N + 1 TIMES, CONFIRMED AT N=3
      iteration   stop_   queue   action
      1           F       1       pop msg -> pending_ -> loop
      2           F       1       pop msg -> pending_ -> loop
      3           F       1       pop msg -> pending_ -> loop
      4           T       0       flushPending() -> break

  If the break condition were stop_ alone, without && empty(), iteration 4 would break with
  the queue already at 0 but pending_ still holding 88 unflushed bytes. The && empty() half of
  the condition is what forces that flush to happen before the thread is allowed to exit.

  --------
  SO THE GENERAL RULE IS ITERATIONS = REQUESTS + 1
  Each request costs exactly one iteration to pop. The final, extra iteration is the one that
  detects the drained, stopping state and flushes whatever pending_ was still holding.

  >> STANDARD NAMES
     N log lines flushed together      -> batch-write (one fprintf, not N)
     turns of the loop for N requests  -> N + 1 iterations
